    Four #[tauri::command] pub async fn:

    history_list(window: tauri::Window, store: tauri::State<'_, ConversationStore>) -> Result<Vec<ConversationSummary>, HistoryError>:
    - assert_main_window(&window)?
    - store.list_conversations().map_err(|e| HistoryError::StorageError(e.to_string())).map(|rows| rows.into_iter().map(|r| ConversationSummary { id: r.id, title: r.title, model: r.model, status: r.status, updated_at: r.updated_at, snippet: None }).collect())

    history_get(window, id: String, conv_store: tauri::State<'_, ConversationStore>, msg_store: tauri::State<'_, MessageStore>) -> Result<ConversationDetail, HistoryError>:
    - assert_main_window(&window)?
    - Get conversation: conv_store.get_conversation(&id).map_err(|e| HistoryError::StorageError(e.to_string()))?.ok_or_else(|| HistoryError::NotFound(id.clone()))?
    - Get messages: msg_store.get_messages(&id).map_err(|e| HistoryError::StorageError(e.to_string()))?
    - Map to ConversationDetail with messages: Vec<MessageSummary>

    history_delete(window, id: String, store: tauri::State<'_, RetentionStore>) -> Result<(), HistoryError>:
    - assert_main_window(&window)?
    - store.delete_conversation(&id).map_err(|e| HistoryError::StorageError(e.to_string()))

    history_search(window, query: String, store: tauri::State<'_, FtsStore>) -> Result<Vec<ConversationSummary>, HistoryError>:
    - assert_main_window(&window)?
    - store.search(&query).map_err(|e| HistoryError::StorageError(e.to_string())).map(|results| results.into_iter().map(|r| ConversationSummary { id: r.id, title: r.title, model: r.model, status: r.status, updated_at: r.updated_at, snippet: Some(r.snippet) }).collect())

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Renderer → IPC | Only main window may call history_* commands; backend-side label enforcement |
| IPC → storage | Typed store methods only; no raw SQL at IPC boundary |
| FTS5 snippet output | Snippet text is trusted backend-generated content; not rendered as HTML |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-06 | Elevation of Privilege | history_* commands called from non-main window | mitigate | assert_main_window fires first in every command handler; returns UNAUTHORIZED_WINDOW before any state access |
| T-03-07 | Information Disclosure | ConversationDetail.messages contains full user prompt content | mitigate | Command restricted to main window; content never logged; HistoryError never contains prompt content; redaction is Phase 4 concern for telemetry |
| T-03-08 | Tampering | history_delete with a conversation_id belonging to another user | accept | Single-user desktop app; no multi-tenancy; delete is authenticated by window label only |
| T-03-09 | Denial of Service | history_search with malformed FTS5 MATCH syntax causing rusqlite error | accept | Returns HistoryError::StorageError to frontend; does not crash process; frontend shows error state |
| T-03-10 | Information Disclosure | HistoryError message field leaks internal storage details | accept | Message field contains rusqlite error string; no secret values; acceptable for local desktop app single-user context |
| T-03-SC | Tampering | cargo build deps | mitigate | No new crate deps added (all stores use existing rusqlite, serde already in Cargo.toml) |
</threat_model>
